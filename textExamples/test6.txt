//if嵌套for和while,首先输入给定数N，然后输入选择1计算求1到N以内所有奇数的和，输入2计算所有偶数的和,否则输出0
main()

{
  int i,N,sum = 0,choice=0;
  N = read();
  choice=read();
  if(choice == 1) {
      for(i=1;i<=N;i=i+1)
      {
         if(i%2 == 1)
	    sum = sum+i;
      }
  }
  else if(choice == 2){
      i=0;
      while(i<N){
      sum = sum + i;
      i = i + 2;
     }
   }
  write(sum);
}
